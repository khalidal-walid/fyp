// count total 2016 and above //
db.try.aggregate(
  [
    {
      $match: {
        year: {
          $gt: 2016
        }
      }
    },
    {
      $count: "total"
    }
  ]
)

// find min max of year //
db.try.aggregate([ 
  { "$group": { 
      "_id": null,
      "max": { "$max": "$year" }, 
      "min": { "$min": "$year" } 
  }}
])

// delete doc based on condition //
db.try.deleteMany({ year : 2009})
